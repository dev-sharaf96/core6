import { Component, OnInit } from '@angular/core';
import { TranslateService } from '@ngx-translate/core';
import { PolicyService } from '../core/services';
import { CommonResponse, INajmStatistics, IPolicy } from '../core/models';

@Component({
  selector: 'app-najm',
  templateUrl: './najm.component.html',
  styleUrls: ['./najm.component.css']
})
export class NajmComponent implements OnInit {
  pieData: any;
  najmStatistics: INajmStatistics;

  pendingCols: any[];
  pendingData: IPolicy[];
  pendingLoading: boolean;
  pendingTotalRecords: number;

  /**
   * @constructor Creates an instance of NajmComponent.
   * @param {PolicyService} _policyService
   * @param {TranslateService} translate
   * @memberof NajmComponent
   */
  constructor(
    private _policyService: PolicyService,
    private translate: TranslateService
  ) {
    translate.get(['najm', 'common']).subscribe(translations => {
      this.pendingCols = [
        { field: 'id', header: translations.common.sN },
        { field: 'policyNo', header: translations.common.policyNumber },
        { field: 'userName', header: translations.common.clientName },
        { field: 'statusCode', header: translations.common.status },
        { field: 'policyFileId', header: translations.najm.refNumber },
        { field: 'companyName', header: translations.common.action },
        { field: 'policyEffectiveDate', header: translations.common.date }
      ];
    });
  }

  ngOnInit() {
    this.pendingLoading = true;
    this.getPieChart();
  }

  /**
   * @method getPieChart()
   * @summary get Pie Chart data and bind it
   *
   * @memberof NajmComponent
   */
  getPieChart() {
    this._policyService.getNajmStatistics().subscribe((response: CommonResponse<INajmStatistics>) => {
      this.najmStatistics = response.data;
      this.pieData = {
        labels: ['Submited', 'Pending', 'Rejected'],
        datasets: [
          {
            data: [
              this.najmStatistics.submited,
              this.najmStatistics.pending,
              this.najmStatistics.rejected
            ],
            backgroundColor: ['#36A2EB', '#FFCE56', '#FF6384'],
            hoverBackgroundColor: ['#36A2EB', '#FFCE56', '#FF6384']
          }
        ]
      };
    });
  }
  /**
   * @method pendingLazyLoad()
   * @summary load pending requests async by pagination
   *
   * @param {*} event
   * @memberof NajmComponent
   */
  pendingLazyLoad(event) {
    this.pendingLoading = true;
    this._policyService
      .getNajmPending(
        `pageIndex=${event.first / event.rows}&pageSize=${event.rows}`
      )
      .subscribe(
        (data: CommonResponse<IPolicy[]>) => {
          this.pendingTotalRecords = data.totalCount;
          this.pendingData = data.data;
          this.pendingLoading = false;

        },
        (error: any) => {
          console.log(error);
        }
      );
  }
}
